63 auto phandle = utils::check_and_open_group(handle, "parameters");
64 ParamDump output;
65
66 // Formats can either be a scalar... or not.
67 std::vector<std::string> formats;
68 output.multi_matrix = false;
69 {
70 auto fhandle = utils::check_and_open_dataset(phandle, "format", H5T_STRING);
71 auto fspace = fhandle.getSpace();
72 if (fspace.getSimpleExtentNdims() == 0) {
73 formats.push_back(utils::load_string(fhandle));
74 } else {
75 if (version < 1001000) {
76 throw std::runtime_error("'format' should be a scalar string in version 1.0");
77 }
78 output.multi_matrix = true;
79 formats = utils::load_string_vector(fhandle);
80 }
81 }
82 output.num_matrices = formats.size();
83
84 auto fihandle = utils::check_and_open_group(phandle, "files");
85 auto nfiles = fihandle.getNumObjs();
86
87 // Checking the runs.
88 std::vector<int> runs;
89 if (output.multi_matrix) {
90 runs = utils::load_integer_vector(phandle, "sample_groups");
91 if (runs.size() != formats.size()) {
92 throw std::runtime_error("'sample_groups' and 'format' should have the same length");
93 }
94
95 int total_files = std::accumulate(runs.begin(), runs.end(), 0);
96 if (total_files != static_cast<int>(nfiles)) {
97 throw std::runtime_error("sum of 'sample_groups' is not equal to the length of 'files'");
98 }
99
100 // Checking that everyone has unique and sorted names.
101 auto names = utils::load_string_vector(phandle, "sample_names");
102 if (names.size() != formats.size()) {
103 throw std::runtime_error("'sample_names' and 'format' should have the same length");
104 }
105
106 if (version >= 2001000) {
107 if (!utils::is_unique_and_sorted(names)) {
108 throw std::runtime_error("duplicated or unsorted values in 'sample_names'");
109 }
110 } else {
111 std::sort(names.begin(), names.end());
112 if (!utils::is_unique_and_sorted(names)) {
113 throw std::runtime_error("duplicated values in 'sample_names'");
114 }
115 }
116 } else {
117 runs.push_back(nfiles);
118 }
119
120 // Checking the files.
121 int sofar = 0;
122 std::vector<std::pair<hsize_t, hsize_t> > bytes;
123 for (size_t r = 0; r < runs.size(); ++r) {
124 auto curf = formats[r];
125 std::vector<std::string> types;
126
127 for (int s = 0; s < runs[r]; ++s, ++sofar) {
128 std::string current = std::to_string(sofar);
129 try {
130 auto curfihandle = utils::check_and_open_group(fihandle, current);
131
132 utils::check_and_open_dataset(curfihandle, "name", H5T_STRING, {});
133 types.push_back(utils::load_string(curfihandle, "type"));
134
135 if (embedded) {
136 bytes.emplace_back(
137 utils::load_integer_scalar<hsize_t>(curfihandle, "offset"),
138 utils::load_integer_scalar<hsize_t>(curfihandle, "size")
139 );
140 } else {
141 utils::check_and_open_dataset(curfihandle, "id", H5T_STRING, {});
142 }
143 } catch (std::exception& e) {
144 throw utils::combine_errors(e, "failed to retrieve information for file " + current);
145 }
146 }
147
148 if (curf == "MatrixMarket") {
149 std::unordered_map<std::string, int> expected;
150 expected["mtx"] = 0;
151 expected["genes"] = 0;
152 expected["annotations"] = 0;
153
154 for (auto t : types) {
155 auto it = expected.find(t);
156 if (it == expected.end()) {
157 throw std::runtime_error("unknown file type '" + t + "' when format is 'MatrixMarket'");
158 }
159 ++(it->second);
160 }
161
162 if (expected["mtx"] != 1) {
163 throw std::runtime_error("expected exactly one 'mtx' file when format is 'MatrixMarket'");
164 }
165 if (expected["genes"] > 1) {
166 throw std::runtime_error("expected no more than one 'genes' file when format is 'MatrixMarket'");
167 }
168 if (expected["annotations"] > 1) {
169 throw std::runtime_error("expected no more than one 'annotation' file when format is 'MatrixMarket'");
170 }
171
172 } else if (curf == "10X") {
173 if (types.size() != 1 || types.front() != "h5") {
174 throw std::runtime_error("expected exactly one 'h5' file when format is '10X'");
175 }
176 } else if (curf == "H5AD") {
177 if (types.size() != 1 || types.front() != "h5") {
178 throw std::runtime_error("expected exactly one 'h5' file when format is 'H5AD'");
179 }
180 }
181 }
182
183 // Checking the files make sense.
184 if (embedded) {
185 hsize_t sofar = 0;
186 for (const auto& b : bytes) {
187 if (b.first != sofar) {
188 throw std::runtime_error("offsets and sizes of 'files' are not sorted and contiguous");
189 }
190 sofar += b.second;
191 }
192 }
193
194 // Checking if there's a batch variable.
195 if (!output.multi_matrix && phandle.exists("sample_factor")) {
196 utils::check_and_open_dataset(phandle, "sample_factor", H5T_STRING, {});
197 output.multi_sample = true;
198 } else {
199 output.multi_sample = output.multi_matrix;
200 }
201
202 // Checking if there's a subset group.
203 output.subset_num = -1;
204
205 if (phandle.exists("subset")) {
206 auto subhandle = utils::check_and_open_group(phandle, "subset");
207
208 if (subhandle.exists("indices")) {
209 auto subidx = utils::load_integer_vector(subhandle, "indices");
210
211 for (auto i : subidx) {
212 if (i < 0) {
213 throw std::runtime_error("indices in 'subset/indices' should be non-negative");
214 }
215 }
216
217 if (!utils::is_unique_and_sorted(subidx)) {
218 throw std::runtime_error("indices in 'subset/indices' should be unique and sorted");
219 }
220
221 output.subset_num = subidx.size();
222 } else {
223 utils::check_and_open_dataset(subhandle, "field", H5T_STRING, {});
224 auto vhandle = utils::check_and_open_dataset(subhandle, "values", H5T_STRING);
225 auto vdims = utils::load_dataset_dimensions(vhandle);
226 if (vdims.size() != 1) {
227 throw std::runtime_error("'subset/values' should be a 1-dimensional string dataset");
228 }
229 }
230 }
231
232 return output;

235inline Details validate_results(const H5::Group& handle, const ParamDump& param_info, int version) {
236 auto rhandle = utils::check_and_open_group(handle, "results");
237 Details output;
238
239 // Pulling out the dimensions and modalities.
240 if (version < 2000000) {
241 output.modalities.push_back("RNA");
242
243 auto dims = utils::load_integer_vector<int>(rhandle, "dimensions");
244 if (dims.size() != 2) {
245 throw std::runtime_error("'dimensions' should be a dataset of length 2");
246 }
247 if (dims[0] < 0 || dims[1] < 0) {
248 throw std::runtime_error("'dimensions' should contain non-negative integers");
249 }
250 output.num_features.push_back(dims[0]);
251 output.num_cells = dims[1];
252 } else {
253 output.num_cells = utils::load_integer_scalar<>(rhandle, "num_cells");
254
255 auto fhandle = utils::check_and_open_group(rhandle, "num_features");
256 size_t nmodals = fhandle.getNumObjs();
257 if (nmodals == 0) {
258 throw std::runtime_error("number of modalities should be positive");
259 }
260
261 for (hsize_t idx = 0; idx < nmodals; ++idx) {
262 std::string modality = fhandle.getObjnameByIdx(idx);
263 output.modalities.push_back(modality);
264 output.num_features.push_back(utils::load_integer_scalar<>(fhandle, modality));
265 }
266 }
267
268 // Checking the number of samples.
269 output.num_samples = 1;
270 if (rhandle.exists("num_samples")) {
271 output.num_samples = utils::load_integer_scalar<int>(rhandle, "num_samples");
272 }
273 if (param_info.multi_matrix) {
274 if (output.num_samples != param_info.num_matrices) {
275 throw std::runtime_error("'num_samples' should be equal to the number of matrices");
276 }
277 } else {
278 if (!param_info.multi_sample && output.num_samples != 1) {
279 throw std::runtime_error("'num_samples' should be 1 for single matrix inputs without 'sample_factor'");
280 }
281 }

283 auto check_unique = [](const std::vector<int>& idx, const std::string& msg) -> void {
284 std::unordered_set<int> used;
285 used.reserve(idx.size());
286 for (auto i : idx) {
287 if (i < 0) {
288 throw std::runtime_error(msg + " contains negative values");
289 } else if (used.find(i) != used.end()) {
290 throw std::runtime_error(msg + " contains duplicate values");
291 }
292 used.insert(i);
293 }
294 };
295
296 if (version >= 2000000) {
297 auto ihandle = utils::check_and_open_group<>(rhandle, "identities");
298 for (size_t m = 0; m < output.modalities.size(); ++m) {
299 auto idx = utils::load_integer_vector<int>(ihandle, output.modalities[m]);
300 if (idx.size() != static_cast<size_t>(output.num_features[m])) {
301 throw std::runtime_error("'identities' for modality '" + output.modalities[m] + "' should have length equal to its number of features");
302 }
303 check_unique(idx, "'identities' for modality '" + output.modalities[m] + "'");
304 }
305
306 } else if (version >= 1002000) {
307 auto idx = utils::load_integer_vector<int>(rhandle, "identities");
308 if (idx.size() != static_cast<size_t>(output.num_features[0])) {
309 throw std::runtime_error("'identities' should have length equal to the number of genes");
310 }
311 check_unique(idx, "'identities'");
312
313 } else {
314 if (param_info.multi_matrix) {
315 auto idx = utils::load_integer_vector<int>(rhandle, "indices");
316 if (idx.size() != static_cast<size_t>(output.num_features[0])) {
317 throw std::runtime_error("'indices' should have length equal to the number of genes");
318 }
319 check_unique(idx, "'indices'");
320
321 } else {
322 auto perms = utils::load_integer_vector<int>(rhandle, "permutation");
323 if (perms.size() != static_cast<size_t>(output.num_features[0])) {
324 throw std::runtime_error("'permutation' should have length equal to the number of genes");
325 }
326
327 // Note that the code below implies that all consecutive entries are present,
328 // otherwise we would see duplicates.
329 std::vector<unsigned char> used(perms.size());
330 for (auto p : perms) {
331 if (p < 0 || static_cast<size_t>(p) >= perms.size()) {
332 throw std::runtime_error("'permutation' contains out-of-range values");
333 } else if (used[p]) {
334 throw std::runtime_error("duplicated index in 'permutation'");
335 }
336 used[p] = 1;
337 }
338 }
339 }
340
341 if (param_info.subset_num > -1) {
342 if (param_info.subset_num != output.num_cells) {
343 throw std::runtime_error("inconsistent number of cells in 'parameters/subset/indices' and 'results/num_cells'");
344 }
345 }
346
347 return output;
348}
