55inline ParamDump validate_parameters(const H5::Group& handle, bool embedded, int version = 1001000) {
56 auto phandle = utils::check_and_open_group(handle, "parameters");
57 ParamDump output;
58
59 // Formats can either be a scalar... or not.
60 std::vector<std::string> formats;
61 output.multi_matrix = false;
62 {
63 auto fhandle = utils::check_and_open_dataset(phandle, "format", H5T_STRING);
64 auto fspace = fhandle.getSpace();
65 if (fspace.getSimpleExtentNdims() == 0) {
66 formats.push_back(utils::load_string(fhandle));
67 } else {
68 if (version < 1001000) {
69 throw std::runtime_error("'format' should be a scalar string in version 1.0");
70 }
71 output.multi_matrix = true;
72 formats = utils::load_string_vector(fhandle);
73 }
74 }
75 output.num_matrices = formats.size();
76
77 auto fihandle = utils::check_and_open_group(phandle, "files");
78 auto nfiles = fihandle.getNumObjs();
79
80 // Checking the runs.
81 std::vector<int> runs;
82 if (output.multi_matrix) {
83 runs = utils::load_integer_vector(phandle, "sample_groups");
84 if (runs.size() != formats.size()) {
85 throw std::runtime_error("'sample_groups' and 'format' should have the same length");
86 }
87
88 int total_files = std::accumulate(runs.begin(), runs.end(), 0);
89 if (total_files != static_cast<int>(nfiles)) {
90 throw std::runtime_error("sum of 'sample_groups' is not equal to the length of 'files'");
91 }
92
93 // Checking that everyone has unique groups.
94 auto names = utils::load_string_vector(phandle, "sample_names");
95 if (names.size() != formats.size()) {
96 throw std::runtime_error("'sample_names' and 'format' should have the same length");
97 }
98
99 std::unordered_set<std::string> stuff;
100 for (auto s : names) {
101 if (stuff.find(s) != stuff.end()) {
102 throw std::runtime_error("duplicated sample name '" + s + "' in 'sample_names'");
103 }
104 stuff.insert(s);
105 }
106 } else {
107 runs.push_back(nfiles);
108 }
109
110 // Checking the files.
111 int sofar = 0;
112 std::vector<std::pair<hsize_t, hsize_t> > bytes;
113 for (size_t r = 0; r < runs.size(); ++r) {
114 auto curf = formats[r];
115 std::vector<std::string> types;
116
117 for (int s = 0; s < runs[r]; ++s, ++sofar) {
118 std::string current = std::to_string(sofar);
119 try {
120 auto curfihandle = utils::check_and_open_group(fihandle, current);
121
122 utils::check_and_open_dataset(curfihandle, "name", H5T_STRING, {});
123 types.push_back(utils::load_string(curfihandle, "type"));
124
125 if (embedded) {
126 bytes.emplace_back(
127 utils::load_integer_scalar<hsize_t>(curfihandle, "offset"),
128 utils::load_integer_scalar<hsize_t>(curfihandle, "size")
129 );
130 } else {
131 utils::check_and_open_dataset(curfihandle, "id", H5T_STRING, {});
132 }
133 } catch (std::exception& e) {
134 throw utils::combine_errors(e, "failed to retrieve information for file " + current);
135 }
136 }
137
138 if (curf == "MatrixMarket") {
139 std::unordered_map<std::string, int> expected;
140 expected["mtx"] = 0;
141 expected["genes"] = 0;
142 expected["annotation"] = 0;
143
144 for (auto t : types) {
145 auto it = expected.find(t);
146 if (it == expected.end()) {
147 throw std::runtime_error("unknown file type '" + t + "' when format is 'MatrixMarket'");
148 }
149 ++(it->second);
150 }
151
152 if (expected["mtx"] != 1) {
153 throw std::runtime_error("expected exactly one 'mtx' file when format is 'MatrixMarket'");
154 }
155 if (expected["genes"] > 1) {
156 throw std::runtime_error("expected no more than one 'genes' file when format is 'MatrixMarket'");
157 }
158 if (expected["annotation"] > 1) {
159 throw std::runtime_error("expected no more than one 'annotation' file when format is 'MatrixMarket'");
160 }
161
162 } else if (curf == "10X") {
163 if (types.size() != 1 || types.front() != "h5") {
164 throw std::runtime_error("expected exactly one 'h5' file when format is '10X'");
165 }
166 } else if (curf == "H5AD") {
167 if (types.size() != 1 || types.front() != "h5") {
168 throw std::runtime_error("expected exactly one 'h5' file when format is 'H5AD'");
169 }
170 }
171 }
172
173 // Checking the files make sense.
174 if (embedded) {
175 std::sort(bytes.begin(), bytes.end());
176 hsize_t sofar = 0;
177 for (const auto& b : bytes) {
178 if (b.first != sofar) {
179 throw std::runtime_error("offsets and sizes of 'files' are not contiguous");
180 }
181 sofar += b.second;
182 }
183 }
184
185 // Checking if there's a batch variable.
186 if (!output.multi_matrix && phandle.exists("sample_factor")) {
187 utils::check_and_open_dataset(phandle, "sample_factor", H5T_STRING, {});
188 output.multi_sample = true;
189 } else {
190 output.multi_sample = output.multi_matrix;
191 }
192
193 return output;

196inline Details validate_results(const H5::Group& handle, const ParamDump& param_info, int version = 1001000) {
197 auto rhandle = utils::check_and_open_group(handle, "results");
198 Details output;
199
200 auto dims = utils::load_integer_vector<int>(rhandle, "dimensions");
201 if (dims.size() != 2) {
202 throw std::runtime_error("'dimensions' should be a dataset of length 2");
203 }
204 if (dims[0] < 0 || dims[1] < 0) {
205 throw std::runtime_error("'dimensions' should contain non-negative integers");
206 }
207 output.num_genes = dims[0];
208 output.num_cells = dims[1];
209
210 // Checking the number of samples.
211 output.num_samples = 1;
212 if (rhandle.exists("num_samples")) {
213 output.num_samples = utils::load_integer_scalar<int>(rhandle, "num_samples");
214 }
215 if (param_info.multi_matrix) {
216 if (output.num_samples != param_info.num_matrices) {
217 throw std::runtime_error("'num_samples' should be equal to the number of matrices");
218 }
219 } else {
220 if (!param_info.multi_sample && output.num_samples != 1) {
221 throw std::runtime_error("'num_samples' should be 1 for single matrix inputs without 'sample_factor'");
222 }
223 }
224
225 if (param_info.multi_matrix) {
226 auto idx = utils::load_integer_vector<int>(rhandle, "indices");
227 if (idx.size() != static_cast<size_t>(dims[0])) {
228 throw std::runtime_error("'indices' should have length equal to the number of genes");
229 }
230 for (auto i : idx) {
231 if (i < 0) {
232 throw std::runtime_error("'indices' contains negative values");
233 }
234 }
235
236 } else {
237 auto perms = utils::load_integer_vector<int>(rhandle, "permutation");
238 if (perms.size() != static_cast<size_t>(dims[0])) {
239 throw std::runtime_error("'permutation' should have length equal to the number of genes");
240 }
241
242 // Note that the code below implies that all consecutive entries are present,
243 // otherwise we would see duplicates.
244 std::vector<unsigned char> used(perms.size());
245 for (auto p : perms) {
246 if (p < 0 || static_cast<size_t>(p) >= perms.size()) {
247 throw std::runtime_error("'permutation' contains out-of-range values");
248 } else if (used[p]) {
249 throw std::runtime_error("duplicated index in 'permutation'");
250 }
251 used[p] = 1;
252 }
253 }
254
255 return output;
256}
